- create a postgres database with following configurations
dbname, username, password, host, port

- create migrations and apply
python manage.py makemigrations weather
python manage.py migrate

- run script to upload weather data using threads
python manage.py upload_weather_data

- drop all tables
DO $$ DECLARE
    r RECORD;
BEGIN
    FOR r IN (SELECT tablename FROM pg_tables WHERE schemaname = current_schema()) LOOP
        EXECUTE 'DROP TABLE IF EXISTS ' || quote_ident(r.tablename) || ' CASCADE';
    END LOOP;
END $$;
